## 长代码块（测试滚动）

```javascript
// 这是一个很长的代码块，用来测试滚动效果
class DataProcessor {
  constructor(data) {
    this.data = data;
    this.processed = false;
  }

  process() {
    if (this.processed) {
      return this.data;
    }

    // 处理数据
    this.data = this.data.map(item => {
      return {
        ...item,
        processed: true,
        timestamp: new Date().toISOString()
      };
    });

    this.processed = true;
    return this.data;
  }

  validate() {
    return this.data.every(item => {
      return item.id && item.name;
    });
  }

  export(format = 'json') {
    if (format === 'json') {
      return JSON.stringify(this.data, null, 2);
    }
    // 其他格式...
    return this.data;
  }
}

// 使用示例
const processor = new DataProcessor([
  { id: 1, name: 'Item 1' },
  { id: 2, name: 'Item 2' },
  { id: 3, name: 'Item 3' }
]);

processor.process();
console.log(processor.export());
```
